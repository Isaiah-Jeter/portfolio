Interactive BPSK Transmitter

This personal project is my rendetion of an Offset Quaditrure Phase Shift Key Transmitter that does both modulation and demodulation of signals. This motivation to make this project came from seeing the recent explosion in funding of hardware communication startup companies. I wanted to learn more about this world since it seems like so much money was being put into it so I started researching. During my research I learned about binary phase key shifting and it seemed like a fundamental concept for communication so I decided to see if I could implement it myself.

I started by just implementing a Binary Phase Shift Key Transmitter (BPSK). After doing some research into how they worked I implemented it pretty quickly and was left feeling a little dissatisfied with how simple it was. I leaned into researching more and saw how BPSK’s were lacking in actual usage due to passsing through the origin during state transitions. Passing through the origin creates a high peak to average pwoer ratio.

import numpy as np
import matplotlib.pyplot as plt
#A class for an Offset Quaditrure Phase Shift Key transmitter that does both modulation and demodulation of signals.
class OQPSK:
    def __init__(self, fs, br, fc):
        self.fs = float(fs) # sampling frequency (Hz)
        self.br = float(br/2) # buadrate (bits/second)
        self.fc = float(fc) # carrier frequency (Hz)
        Ns = int(round(self.fs / self.br))
        if Ns % 2 != 0:
            Ns += 1
        self.Ns = Ns
        self.scale = 1.0 / np.sqrt(2.0)  # energy normalization

    """
    modulates incoming bits into wave signal

    arguments:
        bits : NDArray containing the input bits
        phaseOffset : Int representing the starting phase offset of the signal

    return:
        modulatedSignal : NDArray of the modulated waveform of the input bits
        time : Vector representing the time passed during the modulated waveform
    """
    def modulate(self, bits, phaseOffset: float = 0.0):

        #first we make the incoming data into bit pairs
        if len(bits) % 2 != 0: # if we don't have an even number of bits, we add a 0 to the end
            bits = bits = np.append(bits, 0)

        bitPairs = bits.reshape(-1, 2)

        #change the pairs to I and Q data streams
        iBits = bitPairs[:, 0]
        qBits = bitPairs[:, 1]
        I = np.where(iBits == 0, 1, -1)
        Q = np.where(qBits == 0, 1, -1)
        squareI = np.repeat(I, self.Ns)
        squareQ = np.repeat(Q, self.Ns)

        #Shit the Q signal up by 45 degrees for the offset
        squareQ = np.concatenate([np.zeros(self.Ns // 2), squareQ[:-(self.Ns // 2)]])

        #Generate carrier streams and have one with 90 degreee phase shift
        time = np.arange(len(squareI)) / self.fs
        carrierI = np.cos(2 * np.pi * self.fc * time + phaseOffset)
        carrierQ= np.sin(2 * np.pi * self.fc * time + phaseOffset)

        #Modulate carriers by seperate mixers
        iModulated = squareI * carrierI;
        qModulated = squareQ * carrierQ;

        modulatedSignal = self.scale * (iModulated - qModulated)

        return modulatedSignal, time

    """
    demodulates modulated signal into bits

    arguments:
        modulatedSignal : NDArray of the BPSK-modulated waveform of the input bits

    return:
        demodulatedSignal : Bit array containing the demodulated signal
        time : Vector representing the time passed during the demodulated waveform
    """
    def demodulate(self, modulatedSignal):
        #Generate carrier streams and have one with 90 degreee phase shift
        time = np.arange(len(modulatedSignal)) / self.fs
        carrierI = np.cos(2 * np.pi * self.fc * time)
        carrierQ = np.sin(2 * np.pi * self.fc * time)

        #Get Baseband Signals from modulated signals for Q and I streams
        baseBandI = modulatedSignal * carrierI * (2 * self.scale)
        baseBandQUnaligned = -modulatedSignal * carrierQ * (2 * self.scale)

        # Align I and Q singals from offset
        iBandAligned = baseBandI[:-(self.Ns // 2)]
        qBandAligned = baseBandQUnaligned[(self.Ns // 2):]

        # Use minimum number of full bits
        numBits = min(len(iBandAligned), len(qBandAligned)) // self.Ns
        minIBand = iBandAligned[:numBits*self.Ns].reshape(numBits, self.Ns).sum(axis=1)
        minQBand = qBandAligned[:numBits*self.Ns].reshape(numBits, self.Ns).sum(axis=1)

        #Get bits from I and Q bands
        sampledI = np.where(minIBand >= 0, 1, -1)
        sampledQ = np.where(minQBand >= 0, 1, -1)
        Ibits = (sampledI < 0)
        Qbits = (sampledQ < 0)
        streamPair = np.stack([Ibits, Qbits], axis=1)
        streamPair = streamPair.reshape(-1).astype(np.uint8)

        time = np.arange(len(Ibits) * self.Ns) / fs

        return streamPair, time

bits = np.array([0,0,1,1,0,1,0,0,1])
fs=48000

transmitter = OQPSK(fs, br, fc)
Ns = transmitter.Ns //2
print("Input bits:     ", bits)

outputBits, outputTime = transmitter.demodulate(modulatedSignal)
